    📂 Workflow Snippets

    Python

    import torch

# Simple Linear Layer in PyTorch
class SimpleNet(torch.nn.Module):
    def __init__(self):
        super().__init__()
        self.fc = torch.nn.Linear(3, 1) # 3D coord to 1D intensity

    def forward(self, x):
        return torch.sigmoid(self.fc(x))
